  names[i] = matches[i].split(/ [?=\d]/g)[0];
}

// get just the id
for (var i=0; i < matches.length; i++) {
  m = matches[i].match(/\d\d\d\d/g);
  ids[i] = m[0];
}


// fill the csv of format "id","name";
for (var i=0; i < names.length; i++) {
  csv += '"'+ids[i]+'","'+names[i]+'";';
}
